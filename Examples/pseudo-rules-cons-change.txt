%%% Definitions
word -- string with the ortographic form of the word
pron -- string with the phonetic transcription of the word
new_pron -- string with the modified pronunciation
dict -- dictionary with all CMU words as keys and pron as values

for word, pron in dict:

 %%% Rules after Reis (2006)
 %%% General cases
 if pron has [th]: % e.g. think 
   replace [th] by [f] and append new_pron to dict 
   replace [th] by [s] and append new_pron to dict 
   replace [th] by [t] and append new_pron to dict 
   replace [th] by [tt] and append new_pron to dict 
 elif pron has [dh]: % e.g. there 
   replace [dh] by [d] and append new_pron to dict 

 %%% Rules after Trevisol (2010)
 %%% Devoicing in word final position
 elif [dh$]: % e.g. bathe
   replace [dh] by [th] and append new_pron to dict 

end for
